# Source: orcadeployment/templates/import-dashboard-job.yaml
apiVersion: batch/v1
kind: Job
metadata:
  name: test33-dashboard-import
  namespace: monitoring
spec:
  template:
    spec:
      restartPolicy: Never
      containers:
        - name: importer
          image: curlimages/curl:8.7.1
          command: ["/bin/sh", "-c"]
          args:
            - |
              curl -sS -X POST "http://prometheus-stack-grafana.monitoring.svc.cluster.local/api/dashboards/db" \
                -H "Content-Type: application/json" \
                -u "$GRAFANA_USER:$GRAFANA_PASSWORD" \
                --data-binary @/dashboard/dashboard.json
          env:
            - name: GRAFANA_USER
              valueFrom:
                secretKeyRef:
                  name: grafana-import-credentials
                  key: admin-user
            - name: GRAFANA_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: grafana-import-credentials
                  key: admin-password
          volumeMounts:
            - name: dashboard
              mountPath: /dashboard
      volumes:
        - name: dashboard
          configMap:
            name: test33-dashboard-json

# Source: orcadeployment/templates/canary-rollout.yaml
apiVersion: argoproj.io/v1alpha1
kind: Rollout
metadata:
  name: test33
  namespace: argo-rollouts

spec:
  replicas: 4

  selector:
    matchLabels:
      app: test33

  template:
    metadata:
      labels:
        app: test33
    spec:
      imagePullSecrets:
        - name: docker-registry-creds-secret

      containers:
        - name: test33
          image: "ghcr.io/canary-io/test33:v1"
          ports:
            - name: http
              containerPort: 8080

        - name: nginx
          image: nginx:1.27-alpine
          ports:
            - name: proxy
              containerPort: 8081
            - name: nginx-status
              containerPort: 9113
          volumeMounts:
            - name: nginx-config
              mountPath: /etc/nginx/conf.d

        - name: nginx-exporter
          image: nginx/nginx-prometheus-exporter:1.3.0
          args:
            - --nginx.scrape-uri=http://127.0.0.1:9113/stub_status
            - --web.listen-address=:9114
          ports:
            - name: metrics
              containerPort: 9114

      volumes:
        - name: nginx-config
          configMap:
            name: test33-nginx-config

  strategy:
    canary:
      canaryService: test33-canary
      stableService: test33-stable

      trafficRouting:
        nginx:
          stableIngress: test33-ingress

      steps:
          - setWeight: 25
          - pause: {}
          - setWeight: 50
          - pause: {}
          - setWeight: 75
          - pause: {}
          - setWeight: 100
